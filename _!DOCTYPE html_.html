// --- DOM Elements ---
const quizContainer = document.getElementById('quiz-container');
const recommendationsSection =
document.getElementById('recommendations');
const recommendationResults =
document.getElementById('recommendation-results');
const resetQuizBtn = document.getElementById('reset-quiz');
const quizSection = document.getElementById('quiz');
// --- Quiz Logic ---
quizContainer.addEventListener('click', function(e) {
const card = e.target.closest('.quiz-card');
if (card) {
const genre = card.dataset.genre;
showRecommendations(genre);
}
});
function showRecommendations(genre) {
const recommendedGames = games[genre] || [];
recommendationResults.innerHTML = ''; // Clear previous results
if (recommendedGames.length > 0) {
recommendedGames.forEach((game, index) => {
const card = document.createElement('div');
card.className = 'recommendation-card bg-gray-700 rounded-lg
overflow-hidden shadow-xl';
card.style.animationDelay = `${index * 0.1}s`;
card.innerHTML = `
<img src="${game.image}" alt="${game.title}" class="w-full h-48
object-cover">
<div class="p-6">
<h4 class="text-2xl font-bold mb-2">${game.title}</h4>
<p class="text-gray-300">${game.description}</p>
</div>
`;
recommendationResults.appendChild(card);
});
} else {
recommendationResults.innerHTML = `<p class="text-center col-span-
full">No recommendations found for this genre.</p>`;
}
quizSection.classList.add('hidden');
recommendationsSection.classList.remove('hidden');
recommendationsSection.scrollIntoView({ behavior: 'smooth' });
}
// --- Reset Quiz ---
resetQuizBtn.addEventListener('click', function() {
recommendationsSection.classList.add('hidden');
quizSection.classList.remove('hidden');
quizSection.scrollIntoView({ behavior: 'smooth' });
});
